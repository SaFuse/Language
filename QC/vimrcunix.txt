set encoding=utf-8

syntax on
set notitle
"ターミナル上からの貼り付けを許可
set paste
set ruler
"不可視文字の表示
set list
"タブ文字の表記
set listchars=tab:%-,trail:-,eol:$

set number
set whichwrap=b,s,h,l,<,>,[,]
set cursorline
set virtualedit=block
"括弧の対応をハイライト
set showmatch
set mouse=a
set expandtab
set tabstop=4
set autoindent

highlight Normal ctermbg=black ctermfg=white

set laststatus=2
set scrolloff=4
set statusline=%<%f\ %m
set statusline+=%=[%l/%L,%2c][%3p%%]\ [FORMAT=%{&ff}]\ [ENC=%{&fileencoding}]

nnoremap Q q

nnoremap <Space>h ^
nnoremap <Space>l $

nnoremap O o<Up><Down><Esc>

nnoremap <C-b> :make<Return>
nnoremap <C-u> :make run<Return>
nnoremap <C-l> :make clean<Return>
